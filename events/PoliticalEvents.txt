# Opposition Against Fascism
country_event = {
	id = politics.opposition.1
	title = politics.opposition.1.t
	desc = politics.opposition.1.d
	picture = GFX_report_event_election_vote

	trigger = {
		has_government = fascism
	}

	mean_time_to_happen = {
		days = @opposition_days
		modifier = {
			factor = stability
		}
	}
	
	option = { #
		name = politics.opposition.1.a

		add_political_power = 25
		add_stability = -0.05
		set_temp_variable = { tmp = @opposition_dissent } custom_effect_tooltip = dissent_change add_to_variable = { dissent_base = tmp } recalculate_dissent = yes
	}

	option = {
		name = politics.opposition.1.b

		random_list = {
			100 = { add_popularity = { ideology = fascism popularity = -0.04 } }
			100 = { add_popularity = { ideology = traditionalism popularity = 0.02 } }
			50 = { add_popularity = { ideology = conservatism popularity = 0.02 } }
		}
		add_stability = 0.05
	}
}

# BORDER INFLUENCE EVENTS

add_namespace = politics.border_influence

@border_influence_chance = 41
@border_influence_dissent = 5
# Fascist Country Spillover
country_event = {
	id = politics.border_influence.1
	title = politics.border_influence.1.t
	desc = politics.border_influence.1.d
	picture = GFX_report_event_election_vote

	trigger = {
		NOT = {
			has_government = fascism
		}
		any_neighbor_country = {
			has_government = fascism
		}

		set_temp_variable_to_random = rnd
		set_temp_variable = { chance = 1 }
		divide_temp_variable = { chance = @border_influence_chance }
		check_variable = { rnd < chance }
	}

	is_triggered_only = yes
	
	option = { #
		name = politics.border_influence.1.a

		add_political_power = 10
		add_stability = -0.05
		set_temp_variable = { tmp = @border_influence_dissent } custom_effect_tooltip = dissent_change add_to_variable = { dissent_base = tmp } recalculate_dissent = yes
	}

	option = {
		name = politics.border_influence.1.b

		random_list = {
			100 = { add_popularity = { ideology = ROOT popularity = -0.05 } }
			100 = { add_popularity = { ideology = fascism popularity = 0.03 } }
		}
	}
}
