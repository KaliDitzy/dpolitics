country_event = {
	id = election.1
	title = election.1.t
	desc = election.1.d
	picture = GFX_report_event_election_vote
	
	is_triggered_only = yes

	trigger = {
		NOT = {
			OR = {
				fascism_plurality = yes
				traditionalism_plurality = yes
				conservatism_plurality = yes
				liberalism_plurality = yes
				radicalism_plurality = yes
				social_democracy_plurality = yes
				democratic_socialism_plurality = yes
				bolshevism_plurality = yes
				anarchism_plurality = yes
			}
		}
	}
	
	option = {
		name = election.1.a
		add_stability = -0.1
	}
}
# Fascist Victory
country_event = {
	id = election.2
	title = election.2.t
	desc = {
		text = election.2.d.2
		trigger = {
			fascism > 0.5
		}
	}
	desc = {
		text = election.2.d.1
		trigger = {
			NOT = { fascism > 0.5 }
		}
	}
	picture = GFX_report_event_election_vote
	
	is_triggered_only = yes

	trigger = {
		fascism_plurality = yes
	}
	
	option = {
		trigger = {
			NOT = { fascism > 0.5 }
		}
		name = election.2.a
		set_politics = {
			ruling_party = fascism
			elections_allowed = yes
		}
		set_temp_variable = { popularity_hit = party_popularity@fascism }
		multiply_temp_variable = { popularity_hit = -0.1 }
		add_popularity = {
			ideology = fascism
			popularity = var:popularity_hit
		}
	}
	option = {
		trigger = {
			NOT = { fascism > 0.5 }
		}
		name = election.2.b
		add_stability = -0.05
		set_politics = {
			ruling_party = fascism
			elections_allowed = yes
		}
	}
	
	option = {
		trigger = {
			fascism > 0.5
		}
		name = election.2.c
		add_stability = -0.025
		set_politics = {
			ruling_party = fascism
			elections_allowed = yes
		}
	}
}
